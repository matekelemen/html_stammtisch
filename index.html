Code Formatting


                        template <class T>
                        class OptionalRef
                        {
                        private:
                            using PtrInt = std::uintptr_t;
                        public:
                            OptionalRef() noexcept
                            {
                                mPointer.integer = 0;
                            }

                            explicit OptionalRef(T& p) noexcept
                            {
                                auto i = reinterpret_cast<PtrInt>(&p);
                                mPointer.integer = i | 1;
                            }

                            bool has_value() const noexcept
                            {
                                return mPointer.integer & 1;
                            }

                            T& operator*() const
                            {
                                if (has_value()) {
                                    auto i = mPointer.integer & ~static_cast<PtrInt>(1);
                                    return *reinterpret_cast<T*>(i);
                                } else {
                                    throw std::bad_optional_access();
                                }
                            }
                        private:
                            union {
                                T* pointer;
                                PtrInt integer;
                            } mPointer;
                        };
